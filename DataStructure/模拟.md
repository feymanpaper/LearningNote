#### 498.对角线遍历
```go
func findDiagonalOrder(mat [][]int) []int {
    idx:=0
    m:=len(mat)
    n:=len(mat[0])
    ans:=make([]int, m * n)
    cnt:=0
    for idx < m*n{
        if cnt&1==0{
            for i:=min(cnt, m-1); i>=0; i--{
                col:=cnt-i
                if col>=n{
                    break
                }
                ans[idx] = mat[i][col]
                idx++
            }
        }else{
            for i:=min(cnt, n-1); i>=0; i--{
                row:=cnt-i
                if row>=m{
                    break
                }
                ans[idx] = mat[row][i]
                idx++
            }
        }
        cnt++
    }
    return ans
}
```
