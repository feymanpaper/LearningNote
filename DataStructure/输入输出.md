多⾏输⼊，每⾏两个整数
```go
package main

import "fmt"

func main(){
    var a, b int
    for{
        _,err:=fmt.Scanf("%d %d", &a, &b)
        if err!=nil{
            break
        }
        fmt.Println(a+b)
    }
    return 
}
```
多组数据，每组第⼀⾏为n, 之后输⼊n⾏两个整数
```go
package main
import "fmt"
func main(){
    var n int
    var a, b int
    for{
        _,err:=fmt.Scan(&n)
        if err!=nil{
            break
        }
        for i:=0; i<n; i++{
            _,err=fmt.Scan(&a, &b)
            if err!=nil{
                break
            }
            fmt.Println(a+b)
        }
    }
    return 
}
```

```go
package main

func main(){
    var a, b int
    for{
        _,err:=fmt.Scan(&a, &b)
        if err!=nil{
            break
        }
        if a==0&&b==0{
            break
        }
        fmt.Println(a+b)
    }
    return
```
