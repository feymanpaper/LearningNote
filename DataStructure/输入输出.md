多组n⾏数据，每⾏先输⼊⼀个整数N，然后在同⼀⾏内输⼊
M个整数,每组输出之间输出⼀个空⾏
```go
package main
import "fmt"
func main(){
    var n int
    var nnum int
    var temp int
    for{
        _,err:=fmt.Scan(&n)
        if err!=nil{
            break
        }
        for i:=0; i<n; i++{
            _,err=fmt.Scan(&nnum)
            if err!=nil{
                break
            }
            sum:=0
            for j:=0; j<nnum; j++{
                _,err=fmt.Scan(&temp)
                if err!=nil{
                    break
                }
                sum+=temp
            }
            if i==n-1{
                fmt.Printf("%d\n", sum)
            }else{
                fmt.Printf("%d\n\n", sum)
            }
        }
    }
    return 
}
```

```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

type TreeNode struct {
	Left  *TreeNode
	Right *TreeNode
	Val   int
}

func buildTree(arr []int) *TreeNode {
	n := len(arr)
	tree := make([]*TreeNode, n)
	for i := 0; i < n; i++ {
		if arr[i] != -1 {
			tree[i] = &TreeNode{
				Left:  nil,
				Right: nil,
				Val:   arr[i],
			}
		}
	}
	for i := 0; 2*i+2 < n; i++ {
		if tree[i] != nil {
			tree[i].Left = tree[2*i+1]
			tree[i].Right = tree[2*i+2]
		}
	}
	return tree[0]
}

func inorder(root *TreeNode) {
	if root == nil {
		return
	}
	inorder(root.Left)
	fmt.Print(root.Val, " ")
	inorder(root.Right)
}

func preorder(root *TreeNode) {
	if root == nil {
		return
	}
	fmt.Print(root.Val, " ")
	preorder(root.Left)
	preorder(root.Right)
}

func main() {
	//4 1 6 0 2 5 7 -1 -1 -1 3 -1 -1 -1 8
	sc := bufio.NewScanner(os.Stdin)
	for sc.Scan() {
		data := strings.Split(sc.Text(), " ")
		arr := make([]int, len(data))
		for i := 0; i < len(data); i++ {
			arr[i], _ = strconv.Atoi(data[i])
		}
		root := buildTree(arr)
		inorder(root)
		fmt.Println()
		preorder(root)
	}
}
```
