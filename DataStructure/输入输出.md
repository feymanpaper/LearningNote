多⾏输⼊，每⾏两个整数
```go
package main

import "fmt"

func main(){
    var a, b int
    for{
        _,err:=fmt.Scanf("%d %d", &a, &b)
        if err!=nil{
            break
        }
        fmt.Println(a+b)
    }
    return 
}
```
多组数据，每组第⼀⾏为n, 之后输⼊n⾏两个整数
```go
package main
import "fmt"
func main(){
    var n int
    var a, b int
    for{
        _,err:=fmt.Scan(&n)
        if err!=nil{
            break
        }
        for i:=0; i<n; i++{
            _,err=fmt.Scan(&a, &b)
            if err!=nil{
                break
            }
            fmt.Println(a+b)
        }
    }
    return 
}
```
 若⼲⾏输⼊，每⾏输⼊两个整数，遇到特定条件终⽌
```go
package main
import "fmt"
func main(){
    var a, b int
    for{
        _,err:=fmt.Scan(&a, &b)
        if err!=nil{
            break
        }
        if a==0&&b==0{
            break
        }
        fmt.Println(a+b)
    }
    return 
}
```
若⼲⾏输⼊，遇到0终⽌，每⾏第⼀个数为N，表示本⾏后⾯
有N个数
```go
package main
import "fmt"
func main(){
    var n int
    var temp int
    for{
        _,err:=fmt.Scan(&n)
        if err!=nil||n==0{
            break
        }
        sum:=0
        for i:=0; i<n; i++{
            _,err=fmt.Scan(&temp)
            if err!=nil{
                break
            }
            sum+=temp
        }
        fmt.Println(sum)
    }
    return 
}
```
若⼲⾏输⼊，每⾏包括两个整数a和b，由空格分隔，每⾏输出后接⼀个空⾏
```go
package main
import "fmt"
func main(){
    var a, b int
    for{
        _,err:=fmt.Scan(&a, &b)
        if err!=nil{
            break 
        }
        fmt.Printf("%d\n\n", a+b)
    }
    return 
}
```
多组n⾏数据，每⾏先输⼊⼀个整数N，然后在同⼀⾏内输⼊
M个整数,每组输出之间输出⼀个空⾏
```go
package main
import "fmt"
func main(){
    var n int
    var nnum int
    var temp int
    for{
        _,err:=fmt.Scan(&n)
        if err!=nil{
            break
        }
        for i:=0; i<n; i++{
            _,err=fmt.Scan(&nnum)
            if err!=nil{
                break
            }
            sum:=0
            for j:=0; j<nnum; j++{
                _,err=fmt.Scan(&temp)
                if err!=nil{
                    break
                }
                sum+=temp
            }
            if i==n-1{
                fmt.Printf("%d\n", sum)
            }else{
                fmt.Printf("%d\n\n", sum)
            }
        }
    }
    return 
}
```
