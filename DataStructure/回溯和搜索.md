func permute(nums []int) [][]int {
    n:=len(nums)
    ans:=make([][]int, 0)
    temp:=make([]int, n)
    vis:=make([]bool, n)
    var backtrack func(int)
    backtrack=func(idx int){
        if idx==len(nums){
            t:=make([]int, n)
            //注意要做一次深拷贝
            copy(t, temp)
            ans=append(ans, t)
            return 
        }
        for i:=0; i<n; i++{
            if !vis[i]{
                vis[i]=true
                temp[idx]=nums[i]
                backtrack(idx+1)
                //temp[idx]=-1
                vis[i]=false
            }
        }
    }
    backtrack(0)
    return ans
}

func combinationSum(candidates []int, target int) [][]int {
    sort.Slice(candidates, func(i, j int) bool{
        return candidates[i]<candidates[j]
    })
    n:=len(candidates)
    ans:=make([][]int, 0)
    path:=make([]int, 0)
    sum:=0
    var backtrack func(int)
    backtrack = func(idx int){
        if sum>target{
            return 
        }
        if sum==target{
            t:=make([]int, len(path))
            copy(t, path)
            ans=append(ans, t)
            return
        }
        for i:=idx; i<n; i++{
            if sum+candidates[i]>target{
                break
            }
            sum+=candidates[i]
            path=append(path, candidates[i])
            backtrack(i)
            path=path[:len(path)-1]
            sum-=candidates[i]
        }
        return
    }
    backtrack(0)
    return ans
}

func generateParenthesis(n int) []string {
    ans:=make([]string, 0)

    var backtrack func(step, lcnt, rcnt int)

        if rcnt>lcnt{
            return 
        }
        if lcnt+rcnt==2*n{
            ans=append(ans, string(path))
            return 
        }
        if lcnt<n{
            path[step]='('
            backtrack(step+1, lcnt+1, rcnt)
            //
        }
        if rcnt<n{
            path[step]=')'
            backtrack(step+1, lcnt, rcnt+1)
        }
        return 
    }
    backtrack(0, 0, 0)
    return ans
}

func maxAreaOfIsland(grid [][]int) int {
    m:=len(grid)
    n:=len(grid[0])
    vis:=make([][]bool, m)
    for i:=0; i<m; i++{
        vis[i]=make([]bool, n)
    }
    var dfs func(x, y int) int
    dfs = func(x, y int) int{
        if x<0||x>=m||y<0||y>=n{
            return 0
        }
        if vis[x][y]{
            return 0
        }
        vis[x][y]=true
        if grid[x][y]==1{
            return 1+dfs(x+1, y)+dfs(x-1, y)+dfs(x, y+1)+dfs(x, y-1)
        }
        return 0
    }
    ans:=0
    for i:=0; i<m; i++{
        for j:=0; j<n; j++{
            ans=max(ans, dfs(i, j))
        }
    }
    return ans
}

func numIslands(grid [][]byte) int {

                dfs(i, j)
                ans++
            }
        }
    }
    return ans

func exist(board [][]byte, word string) bool {
    m:=len(board)
    n:=len(board[0])
    vis:=make([][]bool, m)
    for i:=0; i<m; i++{
        for j:=0; j<n; j++{
            vis[i]=make([]bool, n)
        }
    }
    var backtrack func(step, x, y int) bool
    backtrack = func(step, x, y int) bool{
        if step==len(word){
            return true
        }
        if x<0||y<0||x>=m||y>=n{
            return false
        }
        if board[x][y]!=word[step]{
            return false
        }
        if vis[x][y]{
            return false
        }
        vis[x][y]=true
        res:=backtrack(step+1,x-1,y)||backtrack(step+1,x+1,y)||backtrack(step+1,x,y-1)||backtrack(step+1,x,y+1)
        vis[x][y]=false
        return res
    }
    for i:=0; i<m; i++{
        for j:=0; j<n; j++{
            if backtrack(0, i, j){
                return true
            }
        }
    }
    return false
}

func solveNQueens(n int) [][]string {
    ans:=make([][]string, 0)
    var backtrack func(row int)
    place:=make([]int, n) //存储第i行放在第几列
    viscol:=make([]bool, n) //某一列是否放过
    dig:=make([]bool, 2*n-1) //主对角线 r+c相同
    subdig:=make([]bool, 2*n-1) //副对角线 r-c相同
    backtrack = func(row int){
        if row==n{
            board:=make([]string, n)
            for r,c:=range place{
                temp:=strings.Repeat(".", c)+"Q"+strings.Repeat(".", n-c-1)
                board[r]=temp
            }
            ans=append(ans, board)
            return 
        }
        for col:=0; col<n; col++{
            if !viscol[col]&&!dig[row+col]&&!subdig[row-col+n-1]{
                place[row]=col
                viscol[col]=true
                dig[row+col]=true
                subdig[row-col+n-1]=true
                backtrack(row+1)
                viscol[col]=false
                dig[row+col]=false
                subdig[row-col+n-1]=false
                //
            }
        }
    }
    backtrack(0)
    return ans
}
